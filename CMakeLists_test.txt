cmake_minimum_required(VERSION 3.20)
project(test_d3d11_implementation)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Add the core and gfx modules
add_subdirectory(src/core)
add_subdirectory(src/gfx)

# Create the test executable
add_executable(test_d3d11_implementation test_d3d11_implementation.cpp)

# Link against our modules
target_link_libraries(test_d3d11_implementation 
    PRIVATE 
    ve_core 
    ve_gfx
)

# Include directories
target_include_directories(test_d3d11_implementation 
    PRIVATE 
    src/core/include
    src/gfx/include
)
